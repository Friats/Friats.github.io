常识

Opencv读取格式是BGR

BGR三通道值对应颜色深度0-255

灰度图只有一个通道表示亮度

图像大小用矩阵表示

w h，表示像素点多少

1.14M图片=720×547×3×8 bit/8(B)

PNG图片除了RGB外还有个alpha通道，alpha是透明度信息

函数及示例

import cv2

#读取图片
img=cv2.imread('当前路径',1)#读入图像,0灰度图,1参数图片
#img:dtype=uint8,[h,w,c],格式ndarray
img.shape#导出高宽通道数
cv2.imshow('image',img)#显示图像窗口名image
cv2.waitKey(0)#等待函数，这里按下空格即可结束等待
cv2.destroyAllWindows()#销毁所有窗口

#保存图片
cv2.imread('名称.jpg',img,[cv2.IMWRITE_JPEG_QUALITY],0-100)#有损压缩，范围0-100，当然也有PNG格式的
cv2.imread('名称.png',img,[cv2.IMWRITE_COMPRESSION],0-9)
#对于JPG格式,数字越小压缩比越高，质量越差，对于PNG来说，数字越小，压缩比越低



#灰度图
img=cv2.imread('当前路径',cv2.IMREAD_GRAYSCALE)#读入图像
cv2.imwrite('路径名字',img)#保存图片
img.size#像素点数量
img.dtype#数据类

#读取视屏
vc=cv2.VideoCapture('路径名字')
if vc.isOpened():#是否能打开视频
    open.frame = vc.read()#vc.read一帧一帧的读取视频
else:
    open = False
while open:
    ret.frame = vc.read()
    if frame is None:
        break
    if ret == True:
        gray = cv2.cvtColor(frame.cv2.COLOR_BGR2GRAY)#转为灰度
        cv2.imshow('result',gray)
        if cv2.waitKey(10) & 0xFF == 27:#27对应ESC键按下退出，等10ms播放一帧
            break
vc.release()
cv2.destroyALLWindows()

#截取图像部分数据
img=cv2.imread('当前路径')#读入图像
cat=img[0:200,0:200]
cv2.imshow('cat',cat)#显示图像窗口名cat
cv2.waitKey(0)#等待函数，这里按下空格即可结束等待

#颜色通道提取

cur_img = img.copy()

cur_img[:,:,1] = 0
cv2.imshow('cur_img',cur_img)#显示图像窗口名cur_img
cv2.waitKey(0)#等待函数，这里按下空格即可结束等待
cv2.destroyAllWindows()#销毁所有窗口

#边界填充
top_size,bottom_size,left_size,right_size = (50,50,50,50)#上下左右的填充大小值
replicate = cv2.copyMakeBorder(img,top_size,bottom_size,left_size,right_size,borderType=cv2.BORDER_REPLICATE)#复制法，复制最边缘的元素
reflect = cv2.copyMakeBorder(img,top_size,bottom_size,left_size,right_size,cv2.BORDER_REFLECT)#反射法对感兴趣的像素两边进行复制
reflect101 = cv2.copyMakeBorder(img,top_size,bottom_size,left_size,right_size,cv2.BORDER_REFLECT_101)#反射法，以最边缘像素为轴对称
wrap = cv2.copyMakeBorder(img,top_size,bottom_size,left_size,right_size,cv2.BORDER_WRAP)#外包装法cdefgh|abcdefgh|abcdefg
constant = cv2.copyMakeBorder(img,top_size,bottom_size,left_size,right_size,cv2.BORDER_CONSTANT,value=0)#常量法，常数值填充
